156  def control_servo_speed(self, servo_key, target_angle):
157  tolerance = 1.0 # degrees within which we consider the target reached
158  servo = self.servosservos[servo_key]
159  self.get_logger().info(f" servo{servo_key} is moving to new target angle{target_angle}")
160 
161  kp = 0.2 # Proportional gain
162  ki = 0.02 # Integral gain
163  kd = 0.05 # Derivative gain
164  integral = 0
165  last_error = 0
166 
167  while True:
168  with self.locklock:
169  current_angle = self.calc_anglecalc_angle(servo.getPosition())
170  if current_angle is None:
171  raise ValueError("Failed to read position from servo")
172 
173  self.publish_joint_anglespublish_joint_angles(servo_key, current_angle) # Publish the current angles
174  target_angle_minus_current_angle = target_angle - current_angle
175  integral += target_angle_minus_current_angle * 0.2 # Accumulate integral (error * time)
176  derivative = (target_angle_minus_current_angle - last_error) / 0.2 # Derivative (change in error / time)
177  last_error = target_angle_minus_current_angle
178 
179  if np.isclose(current_angle, target_angle, atol=tolerance):
180  servo.wheelRPM(0) # Stop the servo
181  self.get_logger().info(f"target angle {self.completed_movements}/{self.total_movements} reached for {servo_key}")
182  break
183 
184  speed = kp * target_angle_minus_current_angle + ki * integral + kd * derivative
185  servo.wheelRPM(speed)
186 
187  # time.sleep(0.1) # Adjust timing as necessary for more responsive control
188  self.get_logger().info(f"Servo: {servo_key}, speed: {np.round(speed)}, difference: {np.round(target_angle_minus_current_angle)}, target angle: {target_angle}, current angle: {current_angle}")
